429 void USART_CharReception_Callback(void)
430 {
431  __IO uint32_t received_char;
432 
433  /* Read Received character. RXNE flag is cleared by reading of RDR register */
434  received_char = LL_USART_ReceiveData8(USART1);
435 
436  /* Check if received value is corresponding to specific one : S or s */
437  if ((received_char == 'S') || (received_char == 's'))
438  {
439  /* Turn LED2 on/off : Expected character has been received */
440  LED_Toggle();
441  }
442 
443  /* Echo received character on TX */
444  LL_USART_TransmitData8(USART1, received_char);
445 }

508 void Error_Callback(void)
509 {

511
