365 void Configure_RTC(void)
366 {
367  /* Enable Peripheral Clock */
368  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
369 
370  /* RTC peripheral reset */
371  LL_RCC_ClearFlag_RTCRSTREL();
372  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
373  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
374  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
375  LL_RCC_ClearFlag_RTCRSTREL();
376 
377  /* Disable the write protection for RTC registers */
378  LL_RTC_DisableWriteProtection(RTC);
379 
380  /* Set Initialization mode */
381  LL_RTC_EnableInitMode(RTC);
382  /* Check if the Initialization mode is set */
383  while (LL_RTC_IsActiveFlag_INIT(RTC) != 1)
384  {
385  }
386 
387  /* Set prescaler according to source clock */
388  LL_RTC_SetAsynchPrescaler(RTC, RTC_ASYNCH_PREDIV);
389  LL_RTC_SetSynchPrescaler(RTC, RTC_SYNCH_PREDIV);
390 
391  /* Disable wake up timer to modify it */
392  LL_RTC_WAKEUP_Disable(RTC);
393 
394  /* Wait until it is allow to modify wake up reload value */
395  while (LL_RTC_IsActiveFlag_WUTW(RTC) != 1)
396  {
397  }
398 
399  /* Exit Initialization mode */
400  LL_RTC_DisableInitMode(RTC);
401  /* Clear RSF flag */
402  LL_RTC_ClearFlag_RS(RTC);
403  /* Wait the registers to be synchronised */
404  while (LL_RTC_IsActiveFlag_RS(RTC) != 1)
405  {
406  }
407 
408  /* Enable RTC registers write protection */
409  LL_RTC_EnableWriteProtection(RTC);
410 }
