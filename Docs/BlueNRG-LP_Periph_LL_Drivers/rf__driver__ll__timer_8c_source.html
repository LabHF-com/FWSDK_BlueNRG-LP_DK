156 static void _get_calibration_data(TIMER_ContextType *context)
157 {
158  int32_t period,freq;
159  period = TIMER_GET_SLOW_PERIOD;
160 
161  while(period != TIMER_GET_SLOW_PERIOD || period == 0)
162  {
163  period = TIMER_GET_SLOW_PERIOD;
164  }
165 
166  int32_t mul1 = 0x8BCF6 ;
167  int32_t b1 = period >> 8 ;
168  int32_t b2 = period & 0xff ;
169  context->period1 = ((mul1 * b1) + ((b2 * mul1)>>8) + 16 )>>5 ;
170  context->period = period;
171 
172  int32_t mult = 0x753 ;
173  freq = TIMER_GET_SLOW_FREQEUNCY;
174 
175  while(freq != TIMER_GET_SLOW_FREQEUNCY || freq == 0)
176  {
177  freq = TIMER_GET_SLOW_FREQEUNCY;
178  }
179  int32_t a1 = freq >> 6 ;
180  int32_t a2 = a1 * mult ;
181  context->freq1 = (a2 + 128) >> 8 ;
182  context->freq = freq;
183 }

253 static void update_system_time(TIMER_ContextType *context)
254 {
255  uint32_t current_machine_time;
256  current_machine_time = WAKEUP->ABSOLUTE_TIME;
257  uint32_t period = context->last_period1;
258  context->cumulative_time = context->last_calibration_time + \
259  blue_unit_conversion(TIME_ABSDIFF(current_machine_time,
260  context->last_calibration_machine_time),
261  period, MULT64_THR_PERIOD);
262 
263  if ((TIMER_Context.periodic_calibration == 0) && (TIME_ABSDIFF(current_machine_time, context->last_calibration_machine_time) < context->calibration_machine_interval)) {
264  context->cumulative_time += blue_unit_conversion(TIMER_MAX_VALUE, period, MULT64_THR_PERIOD);
265  }
266  context->last_machine_time = current_machine_time;
267  context->last_calibration_machine_time = current_machine_time;
268  context->last_calibration_time = context->cumulative_time;
269  context->last_period1 = context->period1;

690 uint8_t TIMER_SetRadioTimerValue(uint32_t timeout, BOOL event_type, BOOL cal_req)
691 {
692  uint32_t current_time, delay, radio_init_delay, device_delay, rel_timeout, rel_timeout_mt;
693  /*choose the 2nd init duration. Check the event_type and cal. request*/
694  if(event_type == TX)
695  {
696  if(cal_req)
697  {
698  radio_init_delay = TIMER_Context.tx_cal_delay;
699  device_delay = TIMER_Context.tx_cal_delay_st;
700  }
701  else
702  {
703  radio_init_delay = TIMER_Context.tx_no_cal_delay;
704  device_delay = TIMER_Context.tx_cal_delay_st;
705  }
706  }
707  else
708  {
709  if(cal_req)
710  {
711  radio_init_delay = TIMER_Context.rx_cal_delay;
712  device_delay = TIMER_Context.tx_cal_delay_st;
713  }
714  else
715  {
716  radio_init_delay = TIMER_Context.rx_no_cal_delay;
717  device_delay = TIMER_Context.tx_cal_delay_st;
718  }
719  }
720 
721  /* At this point, it is care of the upper layers to guarantee that the timeout represents an absolute time in the future */
722  rel_timeout = timeout - (uint32_t)get_system_time_and_machine(&TIMER_Context,&current_time);
723 
724  rel_timeout_mt = blue_unit_conversion(rel_timeout, TIMER_Context.freq1, MULT64_THR_FREQ);
725 
726  /*Check if the timeout is beyond the wakeup time offset. Then program either the WakeUp timer or the Timer1*/
727  if (rel_timeout > (device_delay + TIMER_Context.hs_startup_time + MARGIN_EXT))
728  {
729  /*The timeout is after the wakeup_time_offset, So it is ok to program the wakeup timer*/
730  delay = rel_timeout_mt - blueglob->WAKEUPINITDELAY - radio_init_delay;
731  WAKEUP->BLUE_WAKEUP_TIME = ((current_time + delay) & TIMER_MAX_VALUE);
732  TIMER_BLUE_SET_REQ_MODE(0);
733  TIMER_DISABLE_TIMER12;
734  TIMER_ENABLE_BLUE_SLEEP_REQ;
735  radio_init_delay += blueglob->WAKEUPINITDELAY;
736  }
737  else
738  {
739  delay = rel_timeout_mt - TIMER_Context.tim12_delay_mt - radio_init_delay;
740  TIMER_SET_TIMER1((current_time + delay) & TIMER_MAX_VALUE);
741  TIMER_DISABLE_WAKEUP_TIMER;
742  radio_init_delay += TIMER_Context.tim12_delay_mt;
743  }
744 
745  TIMER_Context.last_anchor_mt = (current_time + rel_timeout_mt) & TIMER_MAX_VALUE;
746 
747 #if HOST_WAKEUP_FIX_ENABLE
748  blueglob->BYTE4 |= 1<<7;
749  blueglob->BYTE22 = 0xF0;
750  blueglob->BYTE23 = 0xFF;
751 #endif
752 
753  /*Basic low level check with an extra margin of machine units */
754  if ((delay + radio_init_delay) < (radio_init_delay + 5))
755  {
756  TIMER_DISABLE_RADIO_TIMERS;
757  return 1;
758  }
759  TIMER_Context.last_setup_time = blue_unit_conversion(radio_init_delay, TIMER_Context.period1, MULT64_THR_PERIOD);
760  return 0;
761 }
