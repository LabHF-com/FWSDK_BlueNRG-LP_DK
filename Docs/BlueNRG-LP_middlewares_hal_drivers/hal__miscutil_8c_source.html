127 void HAL_CrashHandler(uint32_t msp, uint32_t signature)
128 {
129  volatile uint32_t * crash_info = (volatile uint32_t *)&CrashInfoRam;
130  register uint32_t reg_content;
131  /* Init to zero the crash_info RAM locations */
132  for (reg_content=0; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
133  crash_info[reg_content] = 0;
134  }
135  /* Store Crash Signature */
136  CrashInfoRam.signature = signature;
137  /* Store SP register */
138  CrashInfoRam.SP = msp;
139  for (reg_content=2; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
140  uint32_t *ptr = ((uint32_t *)msp)+(reg_content-2);
141  if ((ptr >= ((uint32_t *) _MEMORY_RAM_BEGIN_)) &&
142  (ptr <= ((uint32_t *) _MEMORY_RAM_END_)))
143  crash_info[reg_content] = *ptr;
144  }
145  NVIC_SystemReset();
146 }

359 void HAL_ToneStart(uint8_t RF_Channel, uint8_t Offset, uint8_t PA_Level)
360 {
361  uint8_t internal_pa_level = PA_Level;
362 
363  /* Calculate the synt divide factor for 16 MHz quarts and +250 kHz offset wrt the channel center frequency
364  * Algorithmically MAK = F_rf*(2^20)/35 = F_rf*(2^15)
365  * With F_rf = (2402+2*RF_Channel)+0.25 MHz
366  * K corresponds to b19-b0 of MAK
367  * A corresponds to b22-b20 of MAK
368  * M corresponds to b27-b23 of MAK
369  */
370  uint32_t kHz_250_scaled = 8192; /* = 0.250*2^20/32 */
371  uint32_t MAK = ((2402UL + 2UL * RF_Channel) << 15);
372 
373  if (Offset == 1U)
374  {
375  MAK += kHz_250_scaled;
376  }
377  else if (Offset == 2U)
378  {
379  MAK -= kHz_250_scaled;
380  }
381  else
382  {
383  /* MISRAC2012-Rule-15.7: explicit else; no action required */
384  }
385 
386  uint8_t M = ((uint8_t)(MAK >> 23) & 0x1FU);
387  uint8_t A = ((uint8_t)(MAK >> 20) & 0x07U);
388  uint32_t K = (MAK & 0x000FFFFFUL) + 1UL;
389 
390  RRM->RADIO_FSM_USR = ((uint32_t)internal_pa_level << 3) | RRM_RADIO_FSM_USR_EN_CALIB_SYNTH_Msk | RRM_RADIO_FSM_USR_EN_CALIB_CBP_Msk;
391  RRM->MOD3_DIG_TST = ((uint32_t)M << 3) | ((uint32_t)A & 0x7UL);
392  RRM->MOD2_DIG_TST = (K >> 12) & 0xFFU;
393  RRM->MOD1_DIG_TST = (K >> 4) & 0xFFU;
394  RRM->MOD0_DIG_TST = ((K & 0x0FU) << 4) | 0x09U;
395 
396  /* Take control of the radio FSM through the test bus */
397  RRM->DTB5_DIG_ENG = 0x09;
398  RRM_TEST = 0x03;
399  RRM->DTB5_DIG_ENG = 0x39;
400  while (RRM->FSM_STATUS_DIG_OUT != 0x04UL);
401  RRM->DTB5_DIG_ENG = 0x3B;
402 }
