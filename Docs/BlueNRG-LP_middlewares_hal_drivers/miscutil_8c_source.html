391 {

393  uint8_t link_status[8];
394  uint16_t link_connection_handles[8];

412 tBleStatus aci_hal_tone_start(uint8_t RF_Channel, uint8_t Offset)
413 {
414  tBleStatus status;
415 
416  if (tone_started) /* already started before */
417  {
418  status = BLE_ERROR_COMMAND_DISALLOWED;
419  }
420  else
421  {
422  if ((RF_Channel >= 40U) || (Offset > 2U))
423  { /* channel ID must be from 0-39 */
424  status = BLE_ERROR_INVALID_HCI_CMD_PARAMS;
425  }
426  else
427  {
428  if (LL_busy() == FALSE)
429  {
430 
431 #if defined(CONFIG_DEVICE_BLUENRG_LPS)
432  /* Set GPIOs for antenna switch in output mode. */
433  LL_GPIO_InitTypeDef GPIO_InitStruct = {
434  .Pin = antenna_conf.Antenna_IDs,
435  .Mode = LL_GPIO_MODE_OUTPUT,
436  .Speed = LL_GPIO_SPEED_FREQ_LOW,
437  .OutputType = LL_GPIO_OUTPUT_PUSHPULL,
438  .Pull = LL_GPIO_PULL_NO,
439  };
440 
441  if(antenna_conf.RF_Activity_Enable)
442  {
443  GPIO_InitStruct.Pin |= LL_GPIO_PIN_7;
444  }
445 
446  /* No need to enable clock of GPIOB peripheral since
447  aci_hal_set_antenna_switch_parameters() must be called to set
448  pin configuration for antenna switch. */
449 
450  LL_GPIO_Init(GPIOB, &GPIO_InitStruct);
451 
452  LL_GPIO_ResetOutputPin(GPIOB, antenna_conf.Antenna_IDs);
453  LL_GPIO_SetOutputPin(GPIOB, antenna_conf.Default_Antenna_ID & antenna_conf.Antenna_IDs);
454 
455  if(antenna_conf.RF_Activity_Enable)
456  {
457  /* Drive RF activity pin to enable the antenna switch. */
458  LL_GPIO_SetOutputPin(GPIOB, LL_GPIO_PIN_7);
459  }
460 #endif
461 
462  HAL_ToneStart(RF_Channel, Offset, pa_level);
463  tone_started = TRUE;
464 
465  status = BLE_STATUS_SUCCESS;
466  }else
467  {
468  status = BLE_ERROR_COMMAND_DISALLOWED;
469  }
470  }
471  }
472 
473  return status;
474 }
